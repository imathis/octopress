import random;



def generate_password():

    alphabet = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNPOPQRSTUVWXYZ'

    number = '1234567890'

    special = '!@#$%^&*()-=_+[]{};:,./\~`?'

    charset = alphabet * 3 + number * 7 + special * 2

    print ''.join([random.choice(charset) for x in xrange(16)])



if __name__ == "__main__":

    generate_password()
